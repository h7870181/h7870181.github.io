function intentPage(pageUrl){
    var pageNum = getCurrentPages().length;
    if (!isOptStrNull(pageUrl)){
        if (pageNum >= 5) {
            wx.redirectTo({
                url: pageUrl,
            })
        } else {
            wx.navigateTo({
                url: pageUrl,
            })
        }
    }
}

格式化时间戳

function formatTimeLong(times) {
    if (times.length == 10) {
        return formatTime(new Date(times * 1000));
    } else if (times.length == 13) {
        return formatTime(new Date(times));
    }
}
function formatTime(date) {
    var year = date.getFullYear()
    var month = date.getMonth() + 1
    var day = date.getDate()
    var hour = date.getHours()
    var minute = date.getMinutes()
    var second = date.getSeconds()
    return [year, month, day].map(formatNumber).join('-') + ' ' + [hour, minute, second].map(formatNumber).join(':')
}

网络请求封装工具类 netUtil.js

function RequestConfig(){
    this.page;
    this.url='';
    this.params={};
    this.netMethod='GET';
    this.callback={
        onPre: function(){},
        onEnd: function(){},
        onSuccess:function (data){},
        onError : function(msgCanShow,code){}
    };
    this.callBack = function(event){
        this.callback = event;
        return this;
    };
    this.setPage = function(event){
        this.page = event;
        return this;
    };
    this.setUrl = function(event){
        this.url = event;
        return this;
    };
    this.setParams = function(event){
        this.params = event;
        return this;
    };
    this.post = function(){
        this.netMethod = 'POST';
        return this;
    };
    this.get = function(){
        this.netMethod = 'GET';
        return this;
    };
    this.send = function(){
        request(this);
    };
}

/**
 * 封装网络请求通用的处理: 参数拼接和结果处理
 * @constant 使用buildRequest().send()来发送请求
 * @param requestConfig
 */
function request(requestConfig){
    //请求前
    if(util.isFunction(requestConfig.callback.onPre) ){
        requestConfig.callback.onPre();
    }
    var url = requestConfig.url;
    var body = requestConfig.params;
    console.log("request data start----->>>>>>>>");
    console.log("url=" + url);
    body = handlerCommonParams(body);
    var contentType ='application/json';
    //post请求逻辑
    if (requestConfig.netMethod =='POST'){
        body = util.json2Form(body);
        contentType = 'application/x-www-form-urlencoded';
    }
    console.log(body);
    console.log("method="+requestConfig.netMethod);
    console.log("<<<<<<<<<-----request data end");
    wx.request({
        url: url,
        method:requestConfig.netMethod,
        header:{'Content-Type':contentType},
        data:body,
        success: function(res) {
            if(util.isFunction(requestConfig.callback.onEnd)){
                requestConfig.callback.onEnd();//请求发出后
            }
            if (res.statusCode = 200) {
                console.log("request success: "+JSON.stringify(res.data));
                requestConfig.callback.onSuccess(res.data);
            }else if(res.statusCode >= 500){
                requestConfig.callback.onError("服务器异常！",res.statusCode);
            }else if(res.statusCode >= 400 && res.statusCode < 500){
                requestConfig.callback.onError("没有找到内容",res.statusCode);
            }else{
                requestConfig.callback.onError("网络请求异常！",res.statusCode);
            }
        },
        fail:function(res){
            console.log("request fail: "+JSON.stringify(res));
            if( util.isFunction(requestConfig.callback.onEnd) ){
                requestConfig.callback.onEnd();//请求发出前
            }
            requestConfig.callback.onError("网络请求异常！",res.statusCode);
        }
    })
}
/**
 * 处理公共参数
 * unionid
 * wechatid
 * timestamp
 * entry
 * uid
 */
function handlerCommonParams(body){
    return body;
}
/**
 * json转表单
 */
function json2Form(json) {
    var str = [];
    for (var p in json) {
        str.push(encodeURIComponent(p) + "=" + encodeURIComponent(json[p]));
    }
    return str.join("&");
}

function isFunction(value) {

        return true;
    } else {
        return false;
    }
}
